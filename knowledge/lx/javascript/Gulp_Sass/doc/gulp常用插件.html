任何插件的使用都要经历以下三步：
1）安装插件：npm
npm install --save gulp-htmlmin
PS：可一次性安装多个插件，插件间用空格隔开

2）引包：require()
var htmlmin = require('gulp-htmlmin')

3）使用：pipe()
gulp.task('htmlmin',function(){
   gulp.src('src/html/*.html')
     .pipe(htmlmin())
     .pipe(gulp.dest('dist/html'));
});


htmml压缩 ：gulp-htmlmin
//参数
var opt = {
  removeComments: true,//清除HTML注释
  collapseWhitespace: true,//压缩HTML
  collapseBooleanAttributes: true,//省略布尔属性的值 <input checked="true"/> ==> <input checked />
  removeEmptyAttributes: true,//删除所有空格作属性值 <input id="" /> ==> <input />
  removeScriptTypeAttributes: true,//删除<script>的type="text/javascript"
  removeStyleLinkTypeAttributes: true,//删除<style>和<link>的type="text/css"
  minifyJS: true,//压缩页面JS
  minifyCSS: true//压缩页面CSS
};

//带参数使用
gulp.src('src/html/*.html')
   .pipe(htmlmin(opt))
   .pipe(gulp.dest('dist/html'));

css压缩：gulp-clean-css
//参数
var opt = {
  advanced: false,//类型：Boolean 默认：true [是否开启高级优化（合并选择器等）]
  compatibility: 'ie7',//保留ie7及以下兼容写法 类型：String 默认：''or'*' [启用兼容模式； 'ie7'：IE7兼容模式，'ie8'：IE8兼容模式，'*'：IE9+兼容模式]
  keepBreaks: true,//类型：Boolean 默认：false [是否保留换行]
  keepSpecialComments: '*'
  //保留所有特殊前缀 当你用autoprefixer生成的浏览器前缀，如果不加这个参数，有可能将会删除你的部分前缀
}

//带参数使用
gulp.src('src/css/*.css')
  .pipe(cssmin(opt))
  .pipe(gulp.dest('dist/css'));

js压缩：gulp-uglify
//参数：
var opt = {
  //mangle: {except: ['require' ,'exports' ,'module' ,'$']}//排除混淆关键字
  mangle:true, //是否修改变量名(混淆)（类型：Boolean，默认：true）
  compress: true,//类型：Boolean 默认：true 是否完全压缩
  preserveComments: 'all' //保留所有注释
}

//带参数使用
gulp.task('jsmin', function () {
  gulp.src(['src/js/*.js', '!src/js/**/{test1,test2}.js'])
    .pipe(uglify(opt))
    .pipe(gulp.dest('dist/js'));
});

合并文件：gulp-concat
//参数
var opt = {
  newLine: ';'  //在合并文件前加入字符
}

//带参数使用
gulp.src('src/js/*.js')
  .pipe(concat('all.js',opt))//all.js为合并后的文件名
  .pipe(gulp.dest('dist/js'));

文件重命名：gulp-rename
//参数
var opt = {
  dirname: "main/text/ciao",//子目录名
  basename: "aloha",//文件名
  prefix: "bonjour-",//前缀
  suffix: "-min",//后缀名
  extname: ".js"//扩展名
}

//带参数使用
gulp.src('src/js/index.js')
  .pipe(rename({ suffix: "-min"}))
  .pipe(gulp.dest('dist/js'));

编译Sass: gulp-sass
gulp.task('sass', function () {
  return gulp.src('./sass/**/*.scss')
    .pipe(sass.sync().on('error', sass.logError))
    .pipe(gulp.dest('./css'));
});

//参数
var opt = {outputStyle: 'compressed'}
outputStyle有四种取值分别为：nested(默认），expanded，compact，compressed。


浏览器同步测试：browser-sync
var browserSync = require('browser-sync');
gulp.task('serve', function() {
  browserSync({
    server: {
      baseDir: ['dist']
    },
  }, function(err, bs) {
    console.log(bs.options.getIn(["urls", "local"]));
  });

  gulp.watch('src/styles/*.less',['style']);
  gulp.watch('src/scripts/*.js',['script']);
  gulp.watch('src/images/*.*',['image']);
  gulp.watch('src/*.html',['html']);
});
